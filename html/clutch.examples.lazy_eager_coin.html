Set Default Proof Using "Type*".

The lazy/eager coins, without tapes
Definition lazy : expr :=
  let: "s" := ref NONEV in
  λ: <>,
    match: !"s" with
    | NONE => let: "b" := flip in
              "s" <- SOME "b" ;;
              "b"
    | SOME "b" => "b"
    end.

Definition eager : expr :=
  let: "b" := flip in
  λ: <>, "b".

An intermediate version of lazy that uses a tape to allow presampling bits during the proof
Definition lazy_with_tape : expr :=
  let: "α" := allocB in
  let: "s" := ref NONEV in
  λ: <>,
    match: !"s" with
    | NONE => let: "b" := flipL "α" in
              "s" <- SOME "b" ;;
              "b"
    | SOME "b" => "b"
    end.
